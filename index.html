Google Sheets koppelen

  2. Geef het de naam DroneLog
  3. Open Extensies → Apps Script
  4. Vervang alle bestaande code door het script hieronder
  5. Klik op Implementeren → Nieuwe implementatie → Web-app
  6. Stel "Wie heeft toegang" in op Iedereen
  7. Kopieer de Web App URL en plak hem hierboven
Apps Script
var HEADERS = [
  'Datum','Tijd','Pilot','Licentie','Drone','Serienummer',
  'Locatie','Vluchtgebied','GPS Lat','GPS Lon','Vluchttype',
  'Duur (min)','Max Hoogte (m)','Batterij Start (%)','Batterij Eind (%)',
  'Weer','Windkracht (Bft)','Opmerkingen','Timestamp'
];

function getOrCreateSheet(ss, name) {
  var sheet = ss.getSheetByName(name);
  if (!sheet) {
    sheet = ss.insertSheet(name);
    sheet.appendRow(HEADERS);
    sheet.getRange(1, 1, 1, HEADERS.length)
      .setBackground('#1a5276')
      .setFontColor('#ffffff')
      .setFontWeight('bold');
    sheet.setFrozenRows(1);
  }
  return sheet;
}

function doPost(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var d  = JSON.parse(e.postData.contents);

  var row = [
    d.datum, d.tijd, d.pilot, d.licentie, d.drone, d.serienummer,
    d.locatie, d.vluchtgebied, d.gpsLat, d.gpsLon, d.doel,
    d.duur, d.maxHoogte, d.batterijStart, d.batterijEind,
    d.weer, d.windkracht, d.opmerkingen, new Date().toISOString()
  ];

  // Tabblad per pilot (naam opschonen voor sheetnaam)
  var pilotName = (d.pilot || 'Onbekend').replace(/[:\\\/\?\*\[\]]/g, '').substring(0, 31);
  getOrCreateSheet(ss, pilotName).appendRow(row);

  // Altijd ook naar het tabblad "Alle vluchten"
  getOrCreateSheet(ss, 'Alle vluchten').appendRow(row);

  return ContentService
    .createTextOutput(JSON.stringify({status:'ok'}))
    .setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) {
  return ContentService
    .createTextOutput(JSON.stringify({status:'ok'}))
    .setMimeType(ContentService.MimeType.JSON);
}
